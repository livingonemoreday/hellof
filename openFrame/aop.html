package com.tab.test;

import java.util.Arrays;

import org.aspectj.lang.*;
import org.aspectj.lang.annotation.*;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
//以下三个注解是基于注解的方式
@Order(3)//指定切面优先级，越小越高。
@Aspect
@Component
public class TestLog {
	//申明切入点表达式
	@Pointcut("execution(* com.tab.service.impl.*.*(int,int))")
	public void declareJoinPointExpression() {}
	
	//聲明该方法为前置通知
	@Before("declareJoinPointExpression()")
	public void beforeMethod(JoinPoint joinpoint) {
		System.out.println("【前置通知】");
		System.out.println("【目标方法名称】：" + joinpoint.getSignature().getName());
		System.out.println("【目标方法参数】：" + Arrays.asList(joinpoint.getArgs()));
	}
	//后置通知，即便目标方法异常也会执行
	@After("declareJoinPointExpression()")
	public void afterMethod() {
		System.out.println("【后置通知】");
	}
	//返回通知，可获取目标方法返回的结果
	@AfterReturning(value="declareJoinPointExpression()",returning="object")
	public void afterReturningMethod(JoinPoint joinPoint, Object object) {
		System.out.println("【返回通知】");
		System.out.println("【返回结果】："+ object);
	}
	//异常通知，目标方法异常时执行，且可以指定出现特定异常时执行
	@AfterThrowing(value="declareJoinPointExpression()",throwing="object")
	public void afterThrowingMethod(JoinPoint joinPoint, Exception object) {
		System.out.println("【返回通知】");
		System.out.println("【异常信息】："+ object.getMessage());
	}
	//环绕通知：需要携带ProceedingJoinPoint类型的参数，该类型参数可以决定目标方法是否执行
	//类似于动态代理的全过程！并且环绕通知必须有返回值，即为目标方法的返回值
	@Around(value="declareJoinPointExpression()")
	public Object aroundMethod(ProceedingJoinPoint pjd) {
		System.out.println("【环绕通知】");
		Object result = null;
		try {
			result = pjd.proceed();
			System.out.println("【环绕通知结果】：" + result);
		} catch (Throwable e) {
			// TODO: handle exception
		}
		return result;
	}
}

package com.tab.test;

import org.aspectj.lang.annotation.*;
import org.springframework.core.annotation.Order;
import org.springframework.stereotype.Component;
@Order(4)
@Aspect
@Component
public class TestAspectOther {
	//TestLog.declareJoinPointExpression() 不同包下 还要加包名
	@Before("TestLog.declareJoinPointExpression()")
	public void method() {
		System.out.println("【二号切面前置通知】 ====================");
	}
}

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
	xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
	xmlns:aop="http://www.springframework.org/schema/aop"
	xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd
		http://www.springframework.org/schema/aop http://www.springframework.org/schema/aop/spring-aop-4.0.xsd">

	
	<bean id="testService" class="com.tab.service.impl.TestServiceImpl"/>
	<bean id="testLog" class="com.tab.test.TestLog"/>
	<bean id="testAspectOther" class="com.tab.test.TestAspectOther"/>
	
	<!-- 配置AOP -->
	<aop:config>
		<!-- 配置切點表達式 -->
		<aop:pointcut expression="execution(* com.tab.service.impl.*.*(int,int))" id="pointcut"/>
		<!-- 配置切面及通知 -->
		<aop:aspect ref="testLog" order="3">
			<aop:before method="beforeMethod" pointcut-ref="pointcut"/>
			<aop:after-returning method="afterReturningMethod" pointcut-ref="pointcut" returning="object"/>
		</aop:aspect>
		<aop:aspect ref="testAspectOther" order="4">
			<aop:before method="method" pointcut-ref="pointcut"/>
		</aop:aspect>
	</aop:config>
</beans>
