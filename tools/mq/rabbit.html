@SpringBootApplication
@RestController
@EnableRabbit
public class RabbitmqApplication {

	public static void main(String[] args) {
		SpringApplication.run(RabbitmqApplication.class, args);
	}

	@Bean
	public MessageConverter messageConverter() {
		// message序列化组件 默认private MessageConverter messageConverter = new SimpleMessageConverter(); RabbitTemplate.java中
		return new Jackson2JsonMessageConverter();
	}

	@RabbitListener(queues = {"first"})
	public void receiveMsg(Dog dog) {
		System.out.println("dog is coming: " + dog);
	}

	@RabbitListener(queues = {"second"})
	public void receiveMsg1(Message msg) {
		System.out.println(new String(msg.getBody()));
		System.out.println(msg.getMessageProperties());
	}
}

@SpringBootTest
class RabbitmqApplicationTests {

	@Autowired
	private RabbitTemplate rabbitTemplate;

	@Autowired
	private AmqpAdmin amqpAdmin; //管理交换器路由队列绑定规则的组件

	@Test
	void fuckingExchange() {
		amqpAdmin.declareExchange(new DirectExchange("fucking.spring.direct"));
		amqpAdmin.declareQueue(new Queue("fourth", true));
		amqpAdmin.declareBinding(new Binding("fourth", Binding.DestinationType.QUEUE, "fucking.spring.direct", "fourth", null));
		rabbitTemplate.convertAndSend("fucking.spring.direct", "fourth", new Dog("旺财 " + LocalDateTime.now(), 22));
	}

	@Test
	void contextLoads() {
		rabbitTemplate.send("fucking.direct", "first", new Message("滚滚长江东逝水，浪花淘尽英雄".getBytes()));
	}

	@Test
	void contextLoads1() {
		HashMap<String, Object> map = new HashMap<>();
		map.put("name", "张三");
		map.put("age", 22);
		rabbitTemplate.convertAndSend("fucking.direct", "second", map);
	}

	@Test
	void contextLoads2() {
		Object second = rabbitTemplate.receiveAndConvert("first");
		System.out.println(second);
	}
}
