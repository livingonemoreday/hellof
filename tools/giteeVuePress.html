node.js、git

>安装vuepress 
$ npm install -g vuepress

初始化

注册gitee并创建仓库

$ cd /f/workspace/vuePress
$ git clone https://gitee.com/xiejuenideai/shangshanruoshui.git
$ cd shangshanruoshui
$ npm init -y //初始化工程->会在project下生成一个package.json文件
or
$ git init;npm init -y

config.js文件配置

module.exports = {
  base: '/shangshanruoshui/',// 更换为自己的仓库名
  title: '问君能有几多愁', 
  description : '恰似一江春水向东流',
  /*网站 favicon 此href之/指向.vuepress/public目录：
  eg:...\docs\.vuepress\public\04.jpeg（图片资源皆放于该文件夹下）*/
  head: [['link', { rel: 'icon', href: `/04.jpeg` }]],
  port: '9999', //项目本地启动端口
  themeConfig : {
   /* 头部导航设置：所有以/开头的/在本地则指向工程根路径/docs，部署到gitee上则值base地址
   以/结尾的路径都需要在该文件夹下创建一个README.md文件，非/结尾的则是直接指向该md文件
   eg:/a/b 则是指向a/b.md文件*/
    nav : [
    //路径/a/ 则为：/docs/a,且a下有个README.md文件
    { text: '淫诗一篇', link: '/a/' },
    { text: '狼吼一曲', link: '/music/' },
    { text: '杰森斯坦森', link: '/jie/' },
    { text: 'dubbo一下', link: 'http://dubbo.apache.org/en-us/' },//外部链接
    { text: 'Concat', items: [
        { text: '百度一下', link: 'https://www.baidu.com' },
        { text: 'spring一下', link: 'https://spring.io/' },
                    ]
    },
    { text: '首页', link: '/' },
          ],
     //侧边栏     
	 sidebar:{
        //头部导航-【淫诗一篇】的侧边栏 有两个一级侧边标题： 曹操，刘禹锡
        '/a/':[
            //一级侧边：曹操， 以下有两个子级侧边：短歌行，龟虽寿，/a/b/1为路径
            //短歌行就是对应docs/a/b/1.md文件
            {title: '曹操', children: [ ['/a/b/1', '短歌行'],
                                        ['/a/b/2', '龟虽寿']
                                      ]
            },
            {title: '刘禹锡', children: [ ['/a/c/1', '陋室铭'],
                                         ['/a/c/2', '乌衣巷']
                                       ]
            },
              ],
        '/music/':[
            //此为仅有一级侧边标题的写法：
            ['/music/liudehua', '忘情水'],
            ['/music/zhangxueyou', '一千个伤心的理由']
                 ],
        '/jie/':[
            ['/jie/01', '非常人贩'],
            ['/jie/02', '死亡飞车']
             ],
    }
  }
}
