    字符分隔

      FUNCTION split_str_to_arr(pi_str IN VARCHAR2
                           ,pi_separator IN VARCHAR2 DEFAULT ','
                           )
  RETURN type_arr
  IS
    l_fcn_nm      VARCHAR2(62) := c_fcn_nm || '.split_str_to_arr';
    reg_str       VARCHAR2(5);
    v_arr         type_arr := type_arr();

  BEGIN
    IF length(pi_separator) <> 1 THEN
      RETURN v_arr;
    END IF;

    reg_str := '[^' || pi_separator || ']+';
    FOR rc IN (
      SELECT REGEXP_SUBSTR(pi_str, reg_str, 1, LEVEL) AS item
        FROM DUAL
      CONNECT BY LEVEL <= REGEXP_COUNT(pi_str, reg_str))
    LOOP
      v_arr.extend;
      v_arr(v_arr.last) := rc.item;
    END LOOP;

    RETURN v_arr;
  EXCEPTION
    WHEN OTHERS THEN
      -- TODO
    RETURN v_arr;
  END;

    SELECT *
  FROM (SELECT T.*,
               ROW_NUMBER() OVER(PARTITION BY POL_NUM, CLM_SMS_STAT, CLM_AGT_CODE ORDER BY CLM_NUM) NM
          FROM TCLM_SMS_DTLS T
         WHERE PROCESS_STAT = 'C')
 WHERE NM = 1;

    分组合并

    WITH Q AS
 (SELECT 1 KEY, 'A' X
    FROM DUAL
  UNION ALL
  SELECT 2 KEY, 'B' X
    FROM DUAL
  UNION ALL
  SELECT 3 KEY, 'A' X
    FROM DUAL
  UNION ALL
  SELECT 4 KEY, 'A' X
    FROM DUAL)
SELECT X,
       XMLAGG(XMLELEMENT(E, KEY || ',').EXTRACT('//text()')),
       RTRIM(XMLAGG(XMLELEMENT(E, KEY || ',').EXTRACT('//text()')))
        -- XMLELEMENT(E, key || ',') => <E>1,</E>
        -- .EXTRACT('//text()')      => 1,
        -- XMLAGG + group by         => 1,4,3,   XMLTYPE 类型
        -- RTRIM                     => 1,4,3,    TEXT 类型
  FROM Q
 GROUP BY X;
