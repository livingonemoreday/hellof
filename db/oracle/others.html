    分组合并

    WITH Q AS
 (SELECT 1 KEY, 'A' X
    FROM DUAL
  UNION ALL
  SELECT 2 KEY, 'B' X
    FROM DUAL
  UNION ALL
  SELECT 3 KEY, 'A' X
    FROM DUAL
  UNION ALL
  SELECT 4 KEY, 'A' X
    FROM DUAL)
SELECT X,
       XMLAGG(XMLELEMENT(E, KEY || ',').EXTRACT('//text()')),
       RTRIM(XMLAGG(XMLELEMENT(E, KEY || ',').EXTRACT('//text()')))
        -- XMLELEMENT(E, key || ',') => <E>1,</E>
        -- .EXTRACT('//text()')      => 1,
        -- XMLAGG + group by         => 1,4,3,   XMLTYPE 类型
        -- RTRIM                     => 1,4,3,    TEXT 类型
  FROM Q
 GROUP BY X;
