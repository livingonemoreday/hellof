标量类型

  • 数值类型
    • NUMBER，可以表示整数和浮点数，NUMBER[(precision[, scale])]，最高精度38位
    • PLS_INTEGER，BINARY_INTEGER，(-2 << 30) - (2 << 30 - 1)，BINARY溢出的时候能为其指派一个NUMBER类型而不至于发生异常，PLS则会发生异常，建议用PLS
    • SIMPLE_INTEGER属于PLS_INTEGER的子类型，取值范围相同，区别为该类型不允许为空，若不需要溢出检查且不可能是空，建议使用该类型。
  • 字符类型
    • CHAR，定长，最长32767字节，如果达不到指定的长度，会在右边空格补齐（=NULL时不会），所以数据库提取数据时可能要做空格处理。
    • VARCHAR2，变长，最长32767字节，如果作为字段存储是4000字节
    • LONG，32760字节，存储字段是2GB
  • 布尔类型，TRUE/FALSE/NULL
  • 日期类型，DATE/TIMESTAMP
  • %TYPE

复合类型

  • RECORD
declare
  type emp_record is record(v_sal       employees.salary%type,
                            v_email     employees.email%type,
                            v_hire_date date
                          );
  v_emp_record emp_record;
begin
  select salary, email, hire_date
    into v_emp_record
    from employees
   where employee_id = 100;

  dbms_output.put_line('薪资为：' || v_emp_record.v_sal
                    || ',邮箱为：' || v_emp_record.v_email
                    || ',雇用日期为: ' || v_emp_record.v_hire_date);
  --exception 
end;

  TYPE type_arr IS TABLE OF VARCHAR2(30);

  FUNCTION split_str_to_arr(pi_str IN VARCHAR2
                           ,pi_separator IN VARCHAR2 DEFAULT ','
                           )
  RETURN type_arr
  IS
    l_fcn_nm      VARCHAR2(62) := c_fcn_nm || '.split_str_to_arr';
    reg_str       VARCHAR2(5);
    v_arr         type_arr := type_arr();

  BEGIN
    IF length(pi_separator) <> 1 THEN
      RETURN v_arr;
    END IF;

    reg_str := '[^' || pi_separator || ']+';
    FOR rc IN (
      SELECT REGEXP_SUBSTR(pi_str, reg_str, 1, LEVEL) AS item
        FROM DUAL
      CONNECT BY LEVEL <= REGEXP_COUNT(pi_str, reg_str))
    LOOP
      v_arr.extend;
      v_arr(v_arr.last) := rc.item;
    END LOOP;

    RETURN v_arr;
  EXCEPTION
    WHEN OTHERS THEN
      -- TODO
    RETURN v_arr;
  END;
