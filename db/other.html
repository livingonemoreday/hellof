横表转纵表

MySql

SELECT
	t.id,
	t.NAME,
	sum( CASE T.SUBJECT WHEN '语文' THEN T.fraction ELSE 0 END ) AS 语文,
	sum( CASE T.SUBJECT WHEN '数学' THEN T.fraction ELSE 0 END ) AS 数学,
	sum( CASE T.SUBJECT WHEN '英语' THEN T.fraction ELSE 0 END ) AS 英语 
FROM
	student_subject_fraction t
GROUP BY t.id

Oracle

SELECT T.FIELD_VALUE1 SERVICE_NAME,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'CLIENT_ID', T.FIELD_DESC_ENG, NULL)) AS CLIENT_ID,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'CLIENT_SECRET', T.FIELD_DESC_ENG, NULL)) AS CLIENT_SECRET,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'GRANT_TYPE', T.FIELD_DESC_ENG, NULL)) AS GRANT_TYPE,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'SCOPE', T.FIELD_DESC_ENG, NULL)) AS SCOPE,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'TOKEN_URL', T.FIELD_DESC_ENG, NULL)) AS TOKEN_URL,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'HTTP_METHOD', T.FIELD_DESC_ENG, NULL)) AS HTTP_METHOD,
       LISTAGG(DECODE(T.FIELD_VALUE2, 'INTERFACE_URL', T.FIELD_DESC_ENG, NULL)) AS INTERFACE_URL
  FROM FIELD_VALUES T
 WHERE FIELD_NAME = 'CLAIM_SERVICE_CONFIG'
 GROUP BY T.FIELD_VALUE1;
