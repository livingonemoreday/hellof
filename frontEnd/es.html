test、match同exec

  • exec为RegExp的方法，返回值为数组，始终是在第一处匹配时就返回，返回的数组个数与子表达式个数有关。若指定了 g，则下次调用exec时，会从上个匹配的lastIndex开始查找。
  • test为RegExp的方法，返回值boolean，g的效果与exec方法中regexp中g的效果一样
  • match为string的方法，返回值为数组
var r1 = new RegExp("a(.)", "g");
var r2 = new RegExp("a(.)");
var r3 = new RegExp("a", "g");
var r4 = new RegExp("a");
var s = "3abc4,5acc6";
let arr1 = r1.exec(s); //['ab', 'b']; 再次調用['ac', 'c']; 再調null; 再調又從頭開始查找匹配
let arr2 = r2.exec(s); //['ab', 'b']; 后续调用结果不变
let arr3 = r3.exec(s); //
let arr4 = r4.exec(s); //

arr1 = s.match(r1); //['ab’, 'ac']
arr2 = s.match(r2); //['ab’, 'b']
arr3 = s.match(r3); //['a', 'a']
arr4 = s.match(r4); //['a']

/* compile 方法把正则表达式编译为内部格式，从而执行得更快。
当使用同一个正则表达式，进行多次使用时（比如在循环中），效果显著
*/
var reg = new RegExp();
reg.compile("a(.)", "g");
let index = s.search(reg); //返回第一个与之匹配的子串开始的位置，没有返回-1。

            如果只有一个参数，从该参数位置截取到最后
   splice()是操作数组本身，
   slice()操作后数组不会改变 */
let arr = [0, 1, 2, 3, 4];
let result = arr.splice(1, 2, 3, 4); //arr结果为[0, 3, 4, 3, 4], result为[1, 2]
let result = arr.splic(1); //arr为[0], result为[1,2,3,4]

let result = arr.slice(1, 3); //[开始下标，结束下标)返回截取的内容, result为[1, 2]

let compare = (a, b) => a >= b ? (a == b ? 0 : 1) : -1;
let arr = [10,1,6,2, 0];
let result = arr.sort(compare); //结果为：[0, 1, 2, 6, 10], arr == result

function Sup(name) { this.name = name };

  • 空函数实现中转
function extend(sub, sup) {
    let f = new Function();
    f.prototype = sup.prototype;
    sub.prototype = new f();
    // 如此实现sub函数的原型对象指向super函数的原型对象和一个F空函数，
    sub.prototype.constructor = sub;
    // 如此可解耦Sup.call(this, name)
    sub.superClass = sup.prototype;
    // 一层保险
    if (sup.prototype.constructor == Object.prototype.constructor) {
        sup.prototype.constructor = sup;
    }
}
function Sup(name) {
    this.name = name,
    eat = function () {
    }
};
Sup.prototype.walk = function () { }
function Sub(name, age) {
    Sub.superClass.constructor.call(this, name); //解耦
    this.age = age;
};
extend(Sub, Sup);
let obj = new Sub('zhangsan', 22);
// 子类未继承到父类的模板
